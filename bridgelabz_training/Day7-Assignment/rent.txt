class CarRental {
    String customerName, carModel;
    int rentalDays;

    CarRental(String c, String m, int d) {
        customerName = c;
        carModel = m;
        rentalDays = d;
    }

    void calculateCost() {
        int costPerDay = 1000;
        int total = rentalDays * costPerDay;
        System.out.println("Customer: " + customerName + ", Car: " + carModel + ", Total Cost: " + total);
    }
}

public class rent {
    public static void main(String[] args) {
        CarRental cr1 = new CarRental("Mike", "SUV", 5);
        cr1.calculateCost();
    }
}
