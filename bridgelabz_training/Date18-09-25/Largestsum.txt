package java1;

public class largestsum {
	    public static void main(String[] args) {
	        int[] arr = {-2, -3, 4, -1, -2, 1, 5, -3};
	        int maxSoFar = arr[0];
	        int currMax = arr[0];

	        for(int i = 1; i < arr.length; i++) {
	            currMax = Math.max(arr[i], currMax + arr[i]);
	            if(currMax > maxSoFar)
	            	maxSoFar = currMax;
	        }
	        System.out.println("Max Sum = " + maxSoFar);
	    }
	}
