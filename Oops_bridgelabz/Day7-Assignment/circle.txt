package java1;

class Circle {
    double radius;

    Circle() {
        this(1.0); 
    }

    Circle(double r) {
        radius = r;
    }
}

public class circle {
    public static void main(String[] args) {
        Circle c1 = new Circle();
        Circle c2 = new Circle(5.5);

        System.out.println("Circle1 radius: " + c1.radius);
        System.out.println("Circle2 radius: " + c2.radius);
    }
}
